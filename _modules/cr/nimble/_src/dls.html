[docs]def solve_on_submatrix(A, columns, b):
    """Solves the problem :math:`A[:, I] x = b` where I is an
    index set of selected columns
    """
    A = A[:, columns]
    x, r_norms, rank, s = lstsq(A, b)
    r = b - A @ x
    return x, r
